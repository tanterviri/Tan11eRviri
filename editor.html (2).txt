  <script>
    const STORAGE_KEY = 'tanter_gallery_urls';

    const form = document.getElementById('galleryForm');
    const saveBtn = document.getElementById('saveBtn');
    const clearBtn = document.getElementById('clearBtn');
    const statusEl = document.getElementById('status');

    const inputs = [];
    const previews = [];

    for (let i = 1; i <= 8; i++) {
      inputs[i] = document.getElementById('slot' + i);
      previews[i] = document.getElementById('preview' + i);
    }

    function updatePreviews() {
      for (let i = 1; i <= 8; i++) {
        const url = inputs[i].value.trim();
        if (url) {
          previews[i].src = url;
        } else {
          previews[i].src = '';
        }
      }
    }

    function loadFromStorage() {
      const json = localStorage.getItem(STORAGE_KEY);
      if (!json) {
        statusEl.textContent = 'В localStorage пока нет сохранённых ссылок.';
        statusEl.className = 'status';
        return;
      }

      try {
        const urls = JSON.parse(json);
        for (let i = 0; i < 8; i++) {
          const value = urls[i] || '';
          inputs[i + 1].value = value;
        }
        updatePreviews();
        statusEl.textContent = 'Данные загружены из localStorage.';
        statusEl.className = 'status ok';
      } catch (e) {
        console.error('Ошибка JSON при чтении localStorage', e);
        statusEl.textContent = 'Ошибка чтения данных из localStorage.';
        statusEl.className = 'status error';
      }
    }

    function saveToStorage() {
      const urls = [];
      for (let i = 1; i <= 8; i++) {
        urls.push(inputs[i].value.trim());
      }

      try {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(urls));
        updatePreviews();
        statusEl.textContent = 'Ссылки сохранены в localStorage. Обнови основной сайт, чтобы увидеть изменения.';
        statusEl.className = 'status ok';
      } catch (e) {
        console.error('Ошибка записи в localStorage', e);
        statusEl.textContent = 'Не удалось сохранить в localStorage.';
        statusEl.className = 'status error';
      }
    }

    function clearAll() {
      for (let i = 1; i <= 8; i++) {
        inputs[i].value = '';
      }
      updatePreviews();
      localStorage.removeItem(STORAGE_KEY);
      statusEl.textContent = 'Все слоты очищены, запись в localStorage удалена.';
      statusEl.className = 'status';
    }

    saveBtn.addEventListener('click', saveToStorage);
    clearBtn.addEventListener('click', clearAll);

    for (let i = 1; i <= 8; i++) {
      inputs[i].addEventListener('input', updatePreviews);
    }

    loadFromStorage();
  </script>
